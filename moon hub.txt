local Players = game:GetService("Players")
local TextChatService = game:GetService("TextChatService")
game:GetService("RunService")

local admins = {
    roblox_user_10080348379 = true,
    Apenas_eu362 = true,
    kanakiwhakO = true
}

local function IsAdmin(player)
    return admins[player.Name] == true
end

local function FindPlayer(playerName)
    if not playerName then
        return nil
    end
    
    local searchName = playerName:lower()
    
    for _, player in pairs(Players:GetPlayers()) do
        if player.Name:lower() == searchName or player.DisplayName:lower() == searchName then
            return player
        end
    end
    
    for _, player in pairs(Players:GetPlayers()) do
        if player.Name:lower():find(searchName) or player.DisplayName:lower():find(searchName) then
            return player
        end
    end
    
    return nil
end

local function KillPlayer(targetPlayer)
    if targetPlayer then
        local character = targetPlayer.Character
        if character then
            local humanoid = character:FindFirstChildOfClass("Humanoid")
            if humanoid then
                humanoid.Health = 0
            end
        end
    else
        return
    end
end

local function KickPlayer(targetPlayer, reason)
    if targetPlayer then
        local success, error = pcall(function()
            targetPlayer:Kick(reason or "Expulso pelo admin")
        end)
        
        if success then
            print("[KICK] Expulsou " .. (targetPlayer.Name or "Desconhecido") .. " Motivo: " .. ((reason == "" or not reason) and "Nenhum" or reason))
        else
            warn("[KICK] Falha ao expulsar " .. (targetPlayer.Name or "Desconhecido") .. ": " .. tostring(error))
        end
    end
end

local function SendChatMessage(message)
    if TextChatService.ChatInputBarConfiguration and TextChatService.ChatInputBarConfiguration.TargetTextChannel then
        TextChatService.ChatInputBarConfiguration.TargetTextChannel:SendAsync(message)
        print("üì© Mensagem enviada no chat: " .. message)
    else
        warn("‚ö†Ô∏è N√£o foi poss√≠vel encontrar o canal de chat geral.")
    end
end

local function BringPlayer(adminPlayer, targetPlayer)
    if adminPlayer and targetPlayer then
        local adminCharacter = adminPlayer.Character
        local targetCharacter = targetPlayer.Character
        
        if adminCharacter and targetCharacter then
            local adminRoot = adminCharacter:FindFirstChild("HumanoidRootPart") or (adminCharacter:FindFirstChild("Torso") or adminCharacter:FindFirstChild("UpperTorso"))
            local targetRoot = targetCharacter:FindFirstChild("HumanoidRootPart") or (targetCharacter:FindFirstChild("Torso") or targetCharacter:FindFirstChild("UpperTorso"))
            
            if adminRoot and targetRoot then
                local positionInFront = adminRoot.Position + adminRoot.CFrame.LookVector * 2
                local newCFrame = CFrame.new(positionInFront, positionInFront + adminRoot.CFrame.LookVector)
                
                local success, error = pcall(function()
                    targetRoot.CFrame = newCFrame
                    
                    if targetRoot:FindFirstChild("BodyVelocity") then
                        targetRoot.BodyVelocity:Destroy()
                    end
                    if targetRoot:FindFirstChild("VectorForce") then
                        targetRoot.VectorForce:Destroy()
                    end
                end)
                
                if not success then
                    warn("[BRING] falha ao trazer " .. tostring(targetPlayer.Name) .. ": " .. tostring(error))
                end
            end
        else
            return
        end
    else
        return
    end
end

local function FlingPlayer(targetPlayer, minPower, maxPower)
    local minVelocity = minPower or 10
    local maxVelocity = maxPower or 50
    
    if targetPlayer then
        local character = targetPlayer.Character
        if character then
            local rootPart = character:FindFirstChild("HumanoidRootPart") or (character:FindFirstChild("Torso") or character:FindFirstChild("UpperTorso"))
            
            if rootPart then
                local randomX = math.random(-100, 100)
                local randomY = math.random(10, 80)
                local randomZ = math.random(-100, 100)
                local direction = Vector3.new(randomX, randomY, randomZ)
                
                if direction.Magnitude == 0 then
                    direction = Vector3.new(0, 50, 0)
                end
                
                local unitDirection = direction.Unit
                local velocity = math.random(minVelocity, maxVelocity)
                
                local bodyVelocity = Instance.new("BodyVelocity")
                bodyVelocity.MaxForce = Vector3.new(1000000, 1000000, 1000000)
                bodyVelocity.Velocity = unitDirection * velocity
                bodyVelocity.P = 1250
                bodyVelocity.Name = "TempFlingBV"
                bodyVelocity.Parent = rootPart
                
                local bodyAngularVelocity = Instance.new("BodyAngularVelocity")
                bodyAngularVelocity.MaxTorque = Vector3.new(1000000, 1000000, 1000000)
                bodyAngularVelocity.AngularVelocity = Vector3.new(math.random(-10, 10), math.random(-10, 10), math.random(-10, 10))
                bodyAngularVelocity.Name = "TempFlingBAV"
                bodyAngularVelocity.Parent = rootPart
                
                spawn(function()
                    wait(0.5)
                    if bodyVelocity and bodyVelocity.Parent then
                        bodyVelocity:Destroy()
                    end
                    if bodyAngularVelocity and bodyAngularVelocity.Parent then
                        bodyAngularVelocity:Destroy()
                    end
                end)
            end
        else
            return
        end
    else
        return
    end
end

local function ProcessCommand(player, message)
    if IsAdmin(player) then
        local command, args = message:match("^%s*(;[%w_]+)%s*(.-)%s*$")
        
        if command then
            local commandLower = command:lower()
            
            if commandLower == ";kill" and args ~= "" then
                local targetPlayer = FindPlayer(args)
                if targetPlayer then
                    KillPlayer(targetPlayer)
                    print("[KILL] " .. player.Name .. " matou " .. targetPlayer.Name)
                end
                
            elseif commandLower == ";kick" and args ~= "" then
                local targetName, reason = args:match("^(%S+)%s*(.-)%s*$")
                local targetPlayer = FindPlayer(targetName)
                
                if targetPlayer then
                    if reason == "" or not reason then
                        reason = nil
                    end
                    KickPlayer(targetPlayer, reason)
                end
                
            elseif commandLower == ";say" and args ~= "" then
                local message = args:match("^(%S+)%s*(.-)%s*$")
                FindPlayer(message)
                SendChatMessage(message)
                
            elseif commandLower == ";bring" and args ~= "" then
                local targetPlayer = FindPlayer(args)
                if targetPlayer then
                    BringPlayer(player, targetPlayer)
                end
                
            elseif commandLower == ";fling" and args ~= "" then
                local targetPlayer = FindPlayer(args)
                if targetPlayer then
                    FlingPlayer(targetPlayer, 10, 50)
                end
            end
        end
    else
        return
    end
end

Players.PlayerAdded:Connect(function(player)
    player.Chatted:Connect(function(message)
        ProcessCommand(player, message)
    end)
end)

for _, player in pairs(Players:GetPlayers()) do
    player.Chatted:Connect(function(message)
        ProcessCommand(player, message)
    end)
end

TextChatService.MessageReceived:Connect(function(messageData)
    local playerName = messageData.TextSource and messageData.TextSource.Name
    local messageText = messageData.Text
    
    if admins[playerName] and messageText == ";v" then
        task.wait(0.5)
        SendChatMessage("AntiMandrakes_team")
    end
end)

        local character = targetPlayer.Character
        if character then
            local humanoid = character:FindFirstChildOfClass("Humanoid")
            if humanoid then
                humanoid.Health = 0
            end
        end
    else
        return
    end
end

local function KickPlayer(targetPlayer, reason)
    if targetPlayer then
        local success, error = pcall(function()
            targetPlayer:Kick(reason or "Expulso pelo admin")
        end)
        
        if success then
            print("[KICK] Expulsou " .. (targetPlayer.Name or "Desconhecido") .. " Motivo: " .. ((reason == "" or not reason) and "Nenhum" or reason))
        else
            warn("[KICK] Falha ao expulsar " .. (targetPlayer.Name or "Desconhecido") .. ": " .. tostring(error))
        end
    end
end

local function SendChatMessage(message)
    if TextChatService.ChatInputBarConfiguration and TextChatService.ChatInputBarConfiguration.TargetTextChannel then
        TextChatService.ChatInputBarConfiguration.TargetTextChannel:SendAsync(message)
        print("üì© Mensagem enviada no chat: " .. message)
    else
        warn("‚ö†Ô∏è N√£o foi poss√≠vel encontrar o canal de chat geral.")
    end
end

local function BringPlayer(adminPlayer, targetPlayer)
    if adminPlayer and targetPlayer then
        local adminCharacter = adminPlayer.Character
        local targetCharacter = targetPlayer.Character
        
        if adminCharacter and targetCharacter then
            local adminRoot = adminCharacter:FindFirstChild("HumanoidRootPart") or (adminCharacter:FindFirstChild("Torso") or adminCharacter:FindFirstChild("UpperTorso"))
            local targetRoot = targetCharacter:FindFirstChild("HumanoidRootPart") or (targetCharacter:FindFirstChild("Torso") or targetCharacter:FindFirstChild("UpperTorso"))
            
            if adminRoot and targetRoot then
                local positionInFront = adminRoot.Position + adminRoot.CFrame.LookVector * 2
                local newCFrame = CFrame.new(positionInFront, positionInFront + adminRoot.CFrame.LookVector)
                
                local success, error = pcall(function()
                    targetRoot.CFrame = newCFrame
                    
                    if targetRoot:FindFirstChild("BodyVelocity") then
                        targetRoot.BodyVelocity:Destroy()
                    end
                    if targetRoot:FindFirstChild("VectorForce") then
                        targetRoot.VectorForce:Destroy()
                    end
                end)
                
                if not success then
                    warn("[BRING] falha ao trazer " .. tostring(targetPlayer.Name) .. ": " .. tostring(error))
                end
            end
        else
            return
        end
    else
        return
    end
end

local function FlingPlayer(targetPlayer, minPower, maxPower)
    local minVelocity = minPower or 10
    local maxVelocity = maxPower or 50
    
    if targetPlayer then
        local character = targetPlayer.Character
        if character then
            local rootPart = character:FindFirstChild("HumanoidRootPart") or (character:FindFirstChild("Torso") or character:FindFirstChild("UpperTorso"))
            
            if rootPart then
                local randomX = math.random(-100, 100)
                local randomY = math.random(10, 80)
                local randomZ = math.random(-100, 100)
                local direction = Vector3.new(randomX, randomY, randomZ)
                
                if direction.Magnitude == 0 then
                    direction = Vector3.new(0, 50, 0)
                end
                
                local unitDirection = direction.Unit
                local velocity = math.random(minVelocity, maxVelocity)
                
                local bodyVelocity = Instance.new("BodyVelocity")
                bodyVelocity.MaxForce = Vector3.new(1000000, 1000000, 1000000)
                bodyVelocity.Velocity = unitDirection * velocity
                bodyVelocity.P = 1250
                bodyVelocity.Name = "TempFlingBV"
                bodyVelocity.Parent = rootPart
                
                local bodyAngularVelocity = Instance.new("BodyAngularVelocity")
                bodyAngularVelocity.MaxTorque = Vector3.new(1000000, 1000000, 1000000)
                bodyAngularVelocity.AngularVelocity = Vector3.new(math.random(-10, 10), math.random(-10, 10), math.random(-10, 10))
                bodyAngularVelocity.Name = "TempFlingBAV"
                bodyAngularVelocity.Parent = rootPart
                
                spawn(function()
                    wait(0.5)
                    if bodyVelocity and bodyVelocity.Parent then
                        bodyVelocity:Destroy()
                    end
                    if bodyAngularVelocity and bodyAngularVelocity.Parent then
                        bodyAngularVelocity:Destroy()
                    end
                end)
            end
        else
            return
        end
    else
        return
    end
end

local function ProcessCommand(player, message)
    if IsAdmin(player) then
        local command, args = message:match("^%s*(;[%w_]+)%s*(.-)%s*$")
        
        if command then
            local commandLower = command:lower()
            
            if commandLower == ";kill" and args ~= "" then
                local targetPlayer = FindPlayer(args)
                if targetPlayer then
                    KillPlayer(targetPlayer)
                    print("[KILL] " .. player.Name .. " matou " .. targetPlayer.Name)
                end
                
            elseif commandLower == ";kick" and args ~= "" then
                local targetName, reason = args:match("^(%S+)%s*(.-)%s*$")
                local targetPlayer = FindPlayer(targetName)
                
                if targetPlayer then
                    if reason == "" or not reason then
                        reason = nil
                    end
                    KickPlayer(targetPlayer, reason)
                end
                
            elseif commandLower == ";say" and args ~= "" then
                local message = args:match("^(%S+)%s*(.-)%s*$")
                FindPlayer(message)
                SendChatMessage(message)
                
            elseif commandLower == ";bring" and args ~= "" then
                local targetPlayer = FindPlayer(args)
                if targetPlayer then
                    BringPlayer(player, targetPlayer)
                end
                
            elseif commandLower == ";fling" and args ~= "" then
                local targetPlayer = FindPlayer(args)
                if targetPlayer then
                    FlingPlayer(targetPlayer, 10, 50)
                end
            end
        end
    else
        return
    end
end

Players.PlayerAdded:Connect(function(player)
    player.Chatted:Connect(function(message)
        ProcessCommand(player, message)
    end)
end)

for _, player in pairs(Players:GetPlayers()) do
    player.Chatted:Connect(function(message)
        ProcessCommand(player, message)
    end)
end

TextChatService.MessageReceived:Connect(function(messageData)
    local playerName = messageData.TextSource and messageData.TextSource.Name
    local messageText = messageData.Text
    
    if admins[playerName] and messageText == ";v" then
        task.wait(0.5)
        SendChatMessage("AntiMandrakes_team")
    end
end)

                spawn(function()
                    wait(0.5)

            return

        return
    end

                end

        end
    else
        return
    end
